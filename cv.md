### Counting duplicate
```
function duplicateCount(text) {
	let textl = text.toLowerCase();
  let obj = {};
  let counter = 0;
  for (i = 0; i < textl.length; i++) {
     	if (!obj[textl[i]]) {
    		obj[textl[i]] = 1;
      } else if (obj[textl[i]] < 2){
        obj[textl[i]] += 1;
        counter++;
      }
  }
  return counter;
} 
```
